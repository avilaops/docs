Upload Simples

use avl_storage::{StorageClient, PutObjectRequest};

#[tokio::main]
async fn main() -> Result<(), Box> {
    let client = StorageClient::connect("https://storage.avila.cloud").await?;

    // Upload de arquivo
    client.put_object(PutObjectRequest {
        bucket: "my-bucket".to_string(),
        key: "documents/report.pdf".to_string(),
        body: std::fs::read("report.pdf")?,
        content_type: Some("application/pdf".to_string()),
        ..Default::default()
    }).await?;

    Ok(())
}

Multipart Upload (Arquivos Grandes)

use avl_storage::multipart::*;

// 1. Iniciar
let init_response = client.initiate_multipart_upload(
    InitiateMultipartUploadRequest {
        bucket: "my-bucket".to_string(),
        key: "large-video.mp4".to_string(),
        ..Default::default()
    }
).await?;

// 2. Upload de partes (5 MB cada)
let file = std::fs::read("large-video.mp4")?;
let chunk_size = 5 * 1024 * 1024;
let mut parts = vec![];

for (i, chunk) in file.chunks(chunk_size).enumerate() {
    let part = client.upload_part(UploadPartRequest {
        bucket: "my-bucket".to_string(),
        key: "large-video.mp4".to_string(),
        upload_id: init_response.upload_id.clone(),
        part_number: (i + 1) as u32,
        body: chunk.to_vec(),
    }).await?;
    parts.push(part);
}

// 3. Completar
client.complete_multipart_upload(CompleteMultipartUploadRequest {
    bucket: "my-bucket".to_string(),
    key: "large-video.mp4".to_string(),
    upload_id: init_response.upload_id,
    parts,
}).await?;
