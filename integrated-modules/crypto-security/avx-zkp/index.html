Exemplo - ZK-SNARK

    // Circuito: provar conhecimento de x tal que x^3 + x + 5 == 35

        &proving_key,

        x,
        35 // Output p√∫blico
    ).expect("Erro ao provar");

        &verifying_key,
        &proof,
        35
