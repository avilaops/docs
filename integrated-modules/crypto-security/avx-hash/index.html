Exemplo - Hash Streaming

use avx_hash::Blake3;
use std::fs::File;
use std::io::Read;

fn hash_file(path: &str) -> Result<[u8; 32], std::io::Error> {
    let mut file = File::open(path)?;
    let mut hasher = Blake3::new();
    let mut buffer = [0u8; 8192];

    loop {
        let n = file.read(&mut buffer)?;
        if n == 0 { break; }
        hasher.update(&buffer[..n]);
    }

    Ok(hasher.finalize())
}
