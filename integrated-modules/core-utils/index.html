use avx_runtime::Runtime;
use avx_events::{EventBus, Event};

#[derive(Event)]
struct UserCreated {
    id: u64,
    name: String,
}

fn main() {
    let rt = Runtime::new().unwrap();
    let bus = EventBus::new();

    // Subscribe to events
    bus.subscribe::<UserCreated>(|event| {
        println!("User created: {}", event.name);
    });

    // Publish event
    rt.spawn(async move {
        bus.publish(UserCreated {
            id: 1,
            name: "Avila".to_string(),
        }).await;
    });

    rt.block_on();
}
