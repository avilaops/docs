Exemplos de Uso

Servidor HTTP Simples

use avx_http::Server;

#[tokio::main]
async fn main() -> Result<()> {
    Server::builder()
        .route("/", |req| async {
            Response::ok().body(b"Hello, AVX!")
        })
        .bind("0.0.0.0:8080")
        .serve()
        .await
}

Gateway com Load Balancing

use avx_gateway::{Gateway, LoadBalancer, Algorithm};

let lb = LoadBalancer::new(Algorithm::RoundRobin)
    .add_backend("http://backend1:8000")
    .add_backend("http://backend2:8000");

Gateway::builder()
    .route_with_lb("/api", lb)
    .with_jwt_auth("secret")
    .build()
    .await?;
