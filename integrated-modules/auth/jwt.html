use avx_jwt::{encode, decode, Algorithm, Header, Claims};

// Criar e assinar JWT
fn criar_token() -> Result<String> {
    let header = Header::new(Algorithm::RS256);
    let claims = Claims {
        sub: "user_123".to_string(),
        exp: 1735689600,
        iat: 1735603200,
        ..Default::default()
    };
    
    let chave_privada = "sua_chave_privada";
    let token = encode(&header, &claims, chave_privada.as_bytes())?;
    Ok(token)
}

// Verificar e decodificar JWT
fn verificar_token(token: &str) -> Result<Claims> {
    let chave_publica = "sua_chave_publica";
    let mut validation = Validation::new(Algorithm::RS256);
    validation.validate_exp = true;
    
    let token_data = decode::<Claims>(
        token,
        chave_publica.as_bytes(),
        &validation
    )?;
    
    Ok(token_data.claims)
}
