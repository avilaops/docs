Quick Start Example

Get started with Email & Messaging:

use avx_email::{Email, SmtpClient};

// Send email
let email = Email::builder()
    .to("user@example.com")
    .subject("Hello")
    .html("
Welcome!
")
    .build()?;

let client = SmtpClient::new("smtp.gmail.com")?;
client.send(email).await?;
