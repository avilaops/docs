üöÄ Getting Started

1. Fork and Clone

# Fork the repository on GitHub first

# Clone your fork
git clone https://github.com/YOUR_USERNAME/arxis-core.git
cd arxis-core

# Add upstream remote
git remote add upstream https://github.com/avilaops/arxis-core.git

2. Set Up Development Environment

# Install Rust (if not already installed)
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

# Install development tools
rustup component add rustfmt clippy rust-analyzer
cargo install cargo-nextest cargo-audit

# Build the project
cargo build

3. Create a Branch

# Update your fork
git fetch upstream
git checkout main
git merge upstream/main

# Create feature branch
git checkout -b feature/your-feature-name

üì¨ Pull Request Process

1. Make Your Changes

  • Write clean, idiomatic Rust code
  • Follow existing code style
  • Add tests for new functionality
  • Update documentation as needed

2. Test Your Changes

# Run tests
cargo nextest run

# Run clippy (linter)
cargo clippy --all-targets --all-features -- -D warnings

# Format code
cargo fmt

# Check for security issues
cargo audit

3. Commit Your Changes

Follow conventional commit format:

# Format: type(scope): description

git commit -m "feat(crypto): add AES-256-GCM encryption"
git commit -m "fix(http): resolve connection pool leak"
git commit -m "docs(readme): update installation instructions"
git commit -m "test(tensor): add GPU computation tests"

Commit types:

  • feat: New feature
  • fix: Bug fix
  • docs: Documentation changes
  • style: Code formatting (no logic changes)
  • refactor: Code refactoring
  • test: Adding or updating tests
  • perf: Performance improvements
  • chore: Maintenance tasks

4. Push and Create PR

# Push to your fork
git push origin feature/your-feature-name

# Then create a Pull Request on GitHub

5. PR Guidelines

  • ‚úÖ Use a clear, descriptive title
  • ‚úÖ Provide detailed description of changes
  • ‚úÖ Reference related issues (e.g., "Fixes #123")
  • ‚úÖ Include screenshots for UI changes
  • ‚úÖ Ensure all CI checks pass
  • ‚úÖ Request review from maintainers
  • ‚úÖ Be responsive to feedback

üíª Coding Standards

Rust Style Guide

Follow the official Rust Style Guide

Naming Conventions

// Types: PascalCase
struct HttpServer { }
enum ErrorKind { }

// Functions/methods: snake_case
fn process_request() { }
fn get_user_by_id() { }

// Constants: SCREAMING_SNAKE_CASE
const MAX_CONNECTIONS: usize = 1000;
const DEFAULT_TIMEOUT: Duration = Duration::from_secs(30);

// Modules: snake_case
mod network_utils;
mod crypto_primitives;

Error Handling

// Use Result for fallible operations
pub fn connect_to_database(url: &str) -> Result {
    // Implementation
}

// Provide context in errors
Err(Error::ConnectionFailed { 
    url: url.to_string(),
    reason: "timeout".to_string()
})

Performance

‚úÖ Testing

Unit Tests

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_encryption() {
        let key = generate_key();
        let plaintext = b"secret data";
        
        let ciphertext = encrypt(&key, plaintext).unwrap();
        let decrypted = decrypt(&key, &ciphertext).unwrap();
        
        assert_eq!(plaintext, decrypted.as_slice());
    }

    #[test]
    #[should_panic]
    fn test_invalid_key() {
        let key = [0u8; 16]; // Invalid key size
        encrypt(&key, b"data").unwrap();
    }
}

Integration Tests

// tests/integration_test.rs
use avx_http::{Server, Client};

#[tokio::test]
async fn test_http_server() {
    let server = Server::bind("127.0.0.1:0").await.unwrap();
    let addr = server.local_addr();
    
    tokio::spawn(async move {
        server.serve(app).await
    });

    let client = Client::new();
    let response = client
        .get(&format!("http://{}/health", addr))
        .send()
        .await
        .unwrap();

    assert_eq!(response.status(), 200);
}

Benchmarks

// benches/crypto_bench.rs
use criterion::{black_box, criterion_group, criterion_main, Criterion};

fn benchmark_sha256(c: &mut Criterion) {
    let data = vec![0u8; 1024];
    
    c.bench_function("sha256_1kb", |b| {
        b.iter(|| {
            avx_crypto::hash::sha256(black_box(&data))
        });
    });
}

criterion_group!(benches, benchmark_sha256);
criterion_main!(benches);
